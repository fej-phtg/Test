<html lang="de">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Jump and Run Spiel</title>

    <style>

        body { margin: 0; overflow: hidden; }

        canvas { background-color: #282828; display: block; margin: 0 auto; }

    </style>

</head>

<body>

    <canvas id="gameCanvas" width="800" height="400"></canvas>

    <script>

        const canvas = document.getElementById('gameCanvas');

        const ctx = canvas.getContext('2d');

        

        const soccerBallImage = new Image();

        soccerBallImage.src = 'https://upload.wikimedia.org/wikipedia/commons/7/70/Soccerball.svg';


        const player = {

            x: 50, y: 300, width: 50, height: 50, color: '#AFCD98', dy: 0,

            gravity: 0.5, jumpPower: 10, isJumping: false

        };


        const obstacle = {

            x: canvas.width, y: 300, width: 50, height: 50, color: 'white', dx: 5

        };


        let score = 0;

        let gameSpeed = 1;


        function drawPlayer() {

            ctx.fillStyle = player.color;

            ctx.beginPath();

            ctx.arc(player.x + player.width / 2, player.y + player.height / 2, player.width / 2, 0, Math.PI * 2);

            ctx.closePath();

            ctx.fill();

            ctx.drawImage(soccerBallImage, player.x, player.y, player.width, player.height);

        }


        function drawObstacle() {

            ctx.fillStyle = obstacle.color;

            ctx.fillRect(obstacle.x, obstacle.y, obstacle.width, obstacle.height);

        }


        function clearCanvas() {

            ctx.clearRect(0, 0, canvas.width, canvas.height);

        }


        function detectCollision() {

            if (player.x < obstacle.x + obstacle.width && player.x + player.width > obstacle.x &&

                player.y < obstacle.y + obstacle.height && player.y + player.height > obstacle.y) {

                alert('Game Over! Dein Score: ' + score);

                resetGame();

            }

        }


        function resetGame() {

            obstacle.x = canvas.width;

            score = 0;

            gameSpeed = 1;

            player.y = 300;

            player.dy = 0;

            player.isJumping = false;

        }


        function updateObstacle() {

            obstacle.x -= obstacle.dx * gameSpeed;

            if (obstacle.x + obstacle.width < 0) {

                obstacle.x = canvas.width;

                score++;

                gameSpeed += 0.1;

            }

        }


        function updatePlayer() {

            if (player.isJumping) {

                player.dy += player.gravity;

                player.y += player.dy;

                if (player.y + player.height >= canvas.height) {

                    player.y = canvas.height - player.height;

                    player.isJumping = false;

                    player.dy = 0;

                }

            }

        }


        function update() {

            clearCanvas();

            drawPlayer();

            drawObstacle();

            updatePlayer();

            updateObstacle();

            detectCollision();

            requestAnimationFrame(update);

        }


        function jump() {

            if (!player.isJumping) {

                player.isJumping = true;

                player.dy = -player.jumpPower;

            }

        }


        document.addEventListener('keydown', (e) => {

            if (e.code === 'Space') jump();

        });


        soccerBallImage.onload = () => {

            update();

        };

    </script>

</body>

</html>
